pub struct ClipScrollTree {
    pub spatial_nodes: Vec<SpatialNode>,
    coord_systems: Vec<CoordinateSystem>,
    pub pending_scroll_offsets: HashMap<ExternalScrollId, (LayoutPoint, ScrollClamping), BuildHasherDefault<FxHasher>>,
    pub pipelines_to_discard: HashSet<PipelineId, BuildHasherDefault<FxHasher>>,
    nodes_to_update: Vec<(SpatialNodeIndex, TransformUpdateState)>,
}

Fields

spatial_nodes: Vec<SpatialNode>

Nodes which determine the positions (offsets and transforms) for primitives and clips.

coord_systems: Vec<CoordinateSystem>

A list of transforms that establish new coordinate systems. Spatial nodes only establish a new coordinate system when they have a transform that is not a simple 2d translation.

pending_scroll_offsets: HashMap<ExternalScrollId, (LayoutPoint, ScrollClamping), BuildHasherDefault<FxHasher>>pipelines_to_discard: HashSet<PipelineId, BuildHasherDefault<FxHasher>>

A set of pipelines which should be discarded the next time this tree is drained.

nodes_to_update: Vec<(SpatialNodeIndex, TransformUpdateState)>

Temporary stack of nodes to update when traversing the tree.
